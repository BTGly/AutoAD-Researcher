# 1. Delta 语义

```python
_EPSILON = 1e-10

def compute_deltas(
    baseline: float,
    variant: float,
    direction: Literal["maximize", "minimize"],
):
    raw_delta = variant - baseline
    if direction == "maximize":
        improvement_delta = raw_delta
    else:
        improvement_delta = baseline - variant

    abs_baseline = abs(baseline)
    if abs_baseline < _EPSILON:
        raw_relative_pct = None
        improvement_relative_pct = None
    else:
        raw_relative_pct = raw_delta / abs_baseline * 100.0
        improvement_relative_pct = improvement_delta / abs_baseline * 100.0

    return raw_delta, improvement_delta, raw_relative_pct, improvement_relative_pct
```

关键语义：
- 分母使用 `abs(baseline)`，避免负 baseline 反转百分比意义
- `abs(baseline) < 1e-10` 时 `relative_pct=None`，不生成 ±inf
- 统计判定（`improved_above_threshold`、`all_seeds_improved` 等）全部基于 `improvement_delta`

## 2.2 PairedMetricObservation

```python
class PairedMetricObservation(BaseModel):
    seed: int
    baseline_source: BaselineMetricSource
    baseline_value: float

    variant_unit_id: str
    variant_id: str
    variant_metric_ref: ArtifactReferenceV2
    variant_value: float

    direction: Literal["maximize", "minimize"]
    raw_delta: float
    improvement_delta: float
    raw_relative_change_pct: float | None
    improvement_relative_change_pct: float | None

    pair_validity_status: Literal["valid", "invalid", "insufficient_evidence"]
    variant_validity_ref: ArtifactReferenceV2
    baseline_validity_ref: ArtifactReferenceV2
    protocol_fingerprint: str

    @model_validator(mode="after")
    def _deltas_recomputed(self):
        recalc_raw, recalc_imp, recalc_raw_pct, recalc_imp_pct = compute_deltas(
            self.baseline_value, self.variant_value, self.direction,
        )
        if abs(self.raw_delta - recalc_raw) > 1e-9:
            raise ValueError("raw_delta mismatch")
        if abs(self.improvement_delta - recalc_imp) > 1e-9:
            raise ValueError("improvement_delta mismatch")
        if recalc_raw_pct is None:
            if self.raw_relative_change_pct is not None:
                raise ValueError("raw_relative_change_pct should be None (baseline≈0)")
        elif self.raw_relative_change_pct is None:
            raise ValueError("raw_relative_change_pct should not be None")
        elif abs(self.raw_relative_change_pct - recalc_raw_pct) > 1e-9:
            raise ValueError("raw_relative_change_pct mismatch")
        if recalc_imp_pct is None:
            if self.improvement_relative_change_pct is not None:
                raise ValueError("improvement_relative_change_pct should be None (baseline≈0)")
        elif self.improvement_relative_change_pct is None:
            raise ValueError("improvement_relative_change_pct should not be None")
        elif abs(self.improvement_relative_change_pct - recalc_imp_pct) > 1e-9:
            raise ValueError("improvement_relative_change_pct mismatch")
        return self
```

### Observation ↔ Metric Artifact 交叉校验（P0-4，service layer）

`PairedMetricObservation` 当前的 `_deltas_recomputed()` validator 仅基于对象内部的 `baseline_value`/`variant_value` 重算 delta，并未解引用 `baseline_source.metric_ref` 和 `variant_metric_ref` 从真实 `ParsedMetric` artifact 中取出 value。同样，`AggregatedMetricComparison` 的 `mean_*` 和 `completed_seed_count` 也未从 observations 重算。

这意味着可构造一组内部一致、但与真实 metric artifacts 不一致的科研结果。

```python
def validate_observation_against_metric_artifacts(
    observation: PairedMetricObservation,
    aggregate_key: AggregatedMetricKey,
    baseline_parsed_metric: ParsedMetric,
    variant_parsed_metric: ParsedMetric,
    baseline_validity: ScientificValidityReport,
    variant_validity: ScientificValidityReport,
) -> None:
    """验证 observation 的 value、seed、metric name、dataset row 与 pair validity 一致。"""
    if observation.seed != baseline_parsed_metric.seed:
        raise ValueError("observation.seed != baseline_metric.seed")
    if observation.seed != variant_parsed_metric.seed:
        raise ValueError("observation.seed != variant_metric.seed")

    if abs(observation.baseline_value - baseline_parsed_metric.primary_value) > 1e-9:
        raise ValueError("baseline_value mismatch with ParsedMetric")
    if abs(observation.variant_value - variant_parsed_metric.primary_value) > 1e-9:
        raise ValueError("variant_value mismatch with ParsedMetric")

    if aggregate_key.metric_name != baseline_parsed_metric.metric_name:
        raise ValueError("metric_name mismatch")
    if aggregate_key.metric_name != variant_parsed_metric.metric_name:
        raise ValueError("variant metric_name mismatch")
    if baseline_parsed_metric.dataset_row != aggregate_key.dataset_row:
        raise ValueError("baseline dataset_row mismatch")
    if variant_parsed_metric.dataset_row != aggregate_key.dataset_row:
        raise ValueError("variant dataset_row mismatch")

    if baseline_parsed_metric.parse_status != "parsed":
        raise ValueError("baseline metric not successfully parsed")
    if variant_parsed_metric.parse_status != "parsed":
        raise ValueError("variant metric not successfully parsed")
    if observation.baseline_validity_ref.sha256 != observation.baseline_source.validity_ref.sha256:
        raise ValueError("baseline_validity_ref must equal baseline_source.validity_ref")

    expected_validity = derive_pair_validity(
        baseline_validity,
        variant_validity,
        baseline_parsed_metric,
        variant_parsed_metric,
    )
    if observation.pair_validity_status != expected_validity:
        raise ValueError("pair_validity_status mismatch")


def derive_pair_validity(
    baseline_validity: ScientificValidityReport,
    variant_validity: ScientificValidityReport,
    baseline_metric: ParsedMetric,
    variant_metric: ParsedMetric,
) -> Literal["valid", "invalid", "insufficient_evidence"]:
    """组合两侧 validity 与 ParsedMetric parse_status，确定 pair-level 有效性。"""
    if baseline_metric.parse_status != "parsed":
        return "invalid"
    if variant_metric.parse_status != "parsed":
        return "invalid"

    statuses = {
        baseline_validity.overall_validity,
        variant_validity.overall_validity,
    }
    if "invalid" in statuses:
        return "invalid"
    if "insufficient_evidence" in statuses:
        return "insufficient_evidence"
    return "valid"


def validate_aggregate_from_observations(
    aggregate: AggregatedMetricComparison,
) -> None:
    """从 paired_observations 重算 mean_*、seed_count、completed_seed_count。

    这些摘要字段不是独立可写事实——必须与 observations 推导值一致。
    """
    valid_obs = [
        o for o in aggregate.paired_observations
        if o.pair_validity_status == "valid"
    ]
    invalid_obs = [
        o for o in aggregate.paired_observations
        if o.pair_validity_status != "valid"
    ]

    if aggregate.completed_seed_count != len(valid_obs):
        raise ValueError("completed_seed_count != valid observations count")
    if aggregate.seed_count != len(aggregate.paired_observations):
        raise ValueError("seed_count != total observations count")

    if invalid_obs and aggregate.comparison_status not in ("degraded", "missing", "invalid"):
        raise ValueError("invalid observations require degraded, missing, or invalid status")

    if valid_obs:
        recalc_mean_base = sum(o.baseline_value for o in valid_obs) / len(valid_obs)
        recalc_mean_var = sum(o.variant_value for o in valid_obs) / len(valid_obs)
        recalc_mean_imp = sum(o.improvement_delta for o in valid_obs) / len(valid_obs)

        if abs(aggregate.mean_baseline - recalc_mean_base) > 1e-9:
            raise ValueError("mean_baseline mismatch")
        if abs(aggregate.mean_variant - recalc_mean_var) > 1e-9:
            raise ValueError("mean_variant mismatch")
        if abs(aggregate.mean_raw_delta - (recalc_mean_var - recalc_mean_base)) > 1e-9:
            raise ValueError("mean_raw_delta mismatch")
        if abs(aggregate.mean_improvement_delta - recalc_mean_imp) > 1e-9:
            raise ValueError("mean_improvement_delta mismatch")
        if aggregate.comparison_status == "missing":
            raise ValueError("comparison_status=missing incompatible with valid observations")
    else:
        if aggregate.mean_baseline is not None:
            raise ValueError("mean_baseline must be None when no valid observations")
        if aggregate.mean_variant is not None:
            raise ValueError("mean_variant must be None when no valid observations")
        if aggregate.mean_raw_delta is not None:
            raise ValueError("mean_raw_delta must be None when no valid observations")
        if aggregate.mean_improvement_delta is not None:
            raise ValueError("mean_improvement_delta must be None when no valid observations")
        if aggregate.comparison_status not in ("missing", "invalid"):
            raise ValueError("no valid observations require comparison_status missing or invalid")

    for obs in aggregate.paired_observations:
        if obs.variant_id != aggregate.aggregate_key.variant_id:
            raise ValueError(
                f"observation variant_id={obs.variant_id} != "
                f"aggregate_key.variant_id={aggregate.aggregate_key.variant_id}"
            )

## 2.5 IdeaSupportConclusion

```python
class IdeaSupportConclusion(str, Enum):
    MULTIPLE_VARIANTS_DESCRIPTIVE = "multiple_variants_descriptive"
    IMPLEMENTATION_SENSITIVE = "implementation_sensitive"
    CONSISTENTLY_SUPPORTED = "consistently_supported"
    SUPPORTED_BY_AT_LEAST_ONE = "supported_by_at_least_one_variant"
    NOT_SUPPORTED_BY_TESTED = "not_supported_by_tested_variants"
    CANNOT_JUDGE = "cannot_judge"
    NOT_SUPPORTED_OR_NOT_DEMONSTRATED = "not_supported_or_not_demonstrated"
    NOT_DEMONSTRATED_OR_PARTIAL = "not_demonstrated_or_partial"

def _make_incomplete_conclusion(
    variant_id: str,
    protocol_seeds: Sequence[int],
) -> VariantScientificConclusion:
    return VariantScientificConclusion(
        variant_id=variant_id,
        conclusion=ScientificConclusion.INCOMPLETE,
        matched_rule_id="missing_variant",
        completed_seed_pairs=[],
        missing_seed_pairs=list(protocol_seeds),
        evidence_refs=[],
    )

def derive_idea_support(
    conclusions: Sequence[VariantScientificConclusion],
    selected_variant_ids: Sequence[str],
    multiple_variant_policy: Literal["descriptive_only", "rank_by_primary_metric"],
    protocol_seeds: Sequence[int],
) -> IdeaSupportConclusion:
    selected = list(selected_variant_ids)

    if not selected:
        return IdeaSupportConclusion.CANNOT_JUDGE

    if len(selected) != len(set(selected)):
        raise ValueError("duplicate selected_variant_ids")

    by_variant: dict[str, VariantScientificConclusion] = {}
    for c in conclusions:
        if c.variant_id in by_variant:
            raise ValueError(f"duplicate conclusion for variant '{c.variant_id}'")
        if c.variant_id not in selected:
            raise ValueError(f"conclusion for unselected variant '{c.variant_id}'")
        by_variant[c.variant_id] = c

    normalized = [
        by_variant.get(vid, _make_incomplete_conclusion(vid, protocol_seeds))
        for vid in selected
    ]

    statuses = [c.conclusion for c in normalized]
    has_beneficial = ScientificConclusion.BENEFICIAL in statuses
    has_worse = ScientificConclusion.WORSE in statuses
    has_equivalent = ScientificConclusion.PRACTICALLY_EQUIVALENT in statuses
    has_mixed = ScientificConclusion.MIXED in statuses
    all_beneficial = all(s == ScientificConclusion.BENEFICIAL for s in statuses)
    all_worse = all(s == ScientificConclusion.WORSE for s in statuses)
    all_incomplete = all(s == ScientificConclusion.INCOMPLETE for s in statuses)

    if multiple_variant_policy == "descriptive_only" and len(normalized) > 1:
        return IdeaSupportConclusion.MULTIPLE_VARIANTS_DESCRIPTIVE

    if has_beneficial and has_worse:
        return IdeaSupportConclusion.IMPLEMENTATION_SENSITIVE
    if all_beneficial:
        return IdeaSupportConclusion.CONSISTENTLY_SUPPORTED
    if has_beneficial and not has_worse:
        return IdeaSupportConclusion.SUPPORTED_BY_AT_LEAST_ONE
    if all_worse:
        return IdeaSupportConclusion.NOT_SUPPORTED_BY_TESTED
    if all_incomplete:
        return IdeaSupportConclusion.CANNOT_JUDGE
    if has_worse and not has_beneficial and (has_equivalent or has_mixed):
        return IdeaSupportConclusion.NOT_SUPPORTED_OR_NOT_DEMONSTRATED
    if not has_beneficial and not has_worse and (has_equivalent or has_mixed):
        return IdeaSupportConclusion.NOT_DEMONSTRATED_OR_PARTIAL
    return IdeaSupportConclusion.CANNOT_JUDGE
```

## 2.6 ReproducibilityInterpretation

```python
class ReplicationPairEvidence(BaseModel):
    """单个 seed 的 pair-level 证据。方向一致性由 improvement_delta 确定性计算。"""
    seed: int
    paired_observation_ref: ArtifactReferenceV2
    improvement_delta: float
    validity_status: Literal["valid", "invalid", "insufficient_evidence"]

class ReplicationGroup(BaseModel):
    group_id: str
    variant_id: str
    pairs: list[ReplicationPairEvidence]
    evidence_refs: list[ArtifactReferenceV2]

    @model_validator(mode="after")
    def _no_duplicate_seeds(self):
        seeds = [p.seed for p in self.pairs]
        if len(seeds) != len(set(seeds)):
            raise ValueError("duplicate seed in replication group")
        return self

class ReproducibilityInterpretation(BaseModel):
    operational_retry_refs: list[ArtifactReferenceV2]
    scientific_replication_groups: list[ReplicationGroup]
    consistent_variant_ids: list[str]
    inconsistent_variant_ids: list[str]
    insufficient_evidence_variant_ids: list[str]
    evidence_refs: list[ArtifactReferenceV2]

    @model_validator(mode="after")
    def _variant_ids_are_disjoint_and_cover(self):
        consistent = set(self.consistent_variant_ids)
        inconsistent = set(self.inconsistent_variant_ids)
        insufficient = set(self.insufficient_evidence_variant_ids)
        group_ids = {g.variant_id for g in self.scientific_replication_groups}

        if consistent & inconsistent or inconsistent & insufficient or consistent & insufficient:
            raise ValueError("variant_id categories must be disjoint")
        if consistent | inconsistent | insufficient != group_ids:
            raise ValueError("all replication group variant_ids must be classified")
        return self


def _determine_reproducibility(
    groups: list[ReplicationGroup],
    statistical_analysis_plan: StatisticalAnalysisPlan,
    registered_seeds: set[int],
) -> ReproducibilityInterpretation:
    """确定性判定规则。

    1. 验证所有 seed 属于预注册集合
    2. valid pairs 数量 < minimum_completed_pairs → insufficient_evidence
    3. 达标 + 所有 valid pairs improvement_delta 同号（>0 或 <0 或在 equivalence margin 内）→ consistent
    4. 达标 + 存在方向冲突 → inconsistent
    5. invalid / insufficient_evidence pair 不计入 completed_pairs
    """
    min_pairs = statistical_analysis_plan.minimum_completed_pairs_per_variant
    margin = statistical_analysis_plan.equivalence_margin or 0.0

    consistent: list[str] = []
    inconsistent: list[str] = []
    insufficient: list[str] = []

    for group in groups:
        for pair in group.pairs:
            if pair.seed not in registered_seeds:
                raise ValueError(
                    f"seed {pair.seed} not in registered seed set for variant {group.variant_id}"
                )

        valid_pairs = [p for p in group.pairs if p.validity_status == "valid"]
        completed_count = len(valid_pairs)

        if completed_count < min_pairs:
            insufficient.append(group.variant_id)
            continue

        deltas = [p.improvement_delta for p in valid_pairs]
        all_positive = all(d > margin for d in deltas)
        all_negative = all(d < -margin for d in deltas)
        all_within = all(abs(d) <= margin for d in deltas)

        if all_positive or all_negative or all_within:
            consistent.append(group.variant_id)
        else:
            inconsistent.append(group.variant_id)

    return ReproducibilityInterpretation(
        operational_retry_refs=[],
        scientific_replication_groups=groups,
        consistent_variant_ids=consistent,
        inconsistent_variant_ids=inconsistent,
        insufficient_evidence_variant_ids=insufficient,
        evidence_refs=[],
    )
```

输出文件：`reproducibility_report.json`。

### ReplicationPairEvidence ↔ PairedMetricObservation 交叉校验（设计建议）

`ReplicationPairEvidence` 保存 `paired_observation_ref`、`improvement_delta`、`validity_status`。需解引用 observation 验证这些字段与 observation 内容一致：

```python
def validate_replication_pair_against_observation(
    pair: ReplicationPairEvidence,
    observation: PairedMetricObservation,
    registered_seeds: set[int],
) -> None:
    """验证 ReplicationPairEvidence 的 field 与 PairedMetricObservation 内容一致。"""

    资源遥测缺失或预算 artifact 不可用时必须标记为 not_assessable。

def determine_budget_assessment(
    variant_id: str,
    budget: ResourceBudget | None,
    usage_aggregate: VariantResourceAggregate | None,
    baseline_aggregate: BaselineResourceAggregate | None,
) -> VariantBudgetAssessment:
    """确定性预算评估。

    规则：
    - 预算 artifact 缺失或身份不匹配 → not_assessable
    - 资源遥测 not_available → not_assessable
    - 遥测 available + 预算 available → 比较
    """
    if budget is None:
        return VariantBudgetAssessment(
            variant_id=variant_id,
            status="not_assessable",
            reason="resource budget artifact not found or identity mismatch",
            resource_budget_ref=None,
            resource_usage_refs=[],
        )
    if usage_aggregate is None or usage_aggregate.measurement_status == "not_available":
        return VariantBudgetAssessment(
            variant_id=variant_id,
            status="not_assessable",
            reason="measurement_status=not_available",
            resource_budget_ref=budget.budget_ref,  # budget artifact 存在，引用其正式 ref
            resource_usage_refs=usage_aggregate.attempt_report_refs if usage_aggregate else [],
        )
    status = compare_usage_with_budget(
        usage_aggregate,
        baseline_aggregate,
        budget,
    )
    return VariantBudgetAssessment(
        variant_id=variant_id,
        status=status,
        reason=derive_budget_reason(
            variant_id,
            usage_aggregate,
            baseline_aggregate,
            budget,
            status,
        ),
        resource_budget_ref=budget.budget_ref,
        resource_usage_refs=usage_aggregate.attempt_report_refs,
    )

class NextAction(str, Enum):
    CONTINUE_SAME_IDEA = "continue_same_idea"
    SWITCH_IDEA = "switch_idea"
    REPLAN_EXPERIMENT = "replan_experiment"
    STOP = "stop"

_LEGAL_ACTION_STAGES: dict[NextAction, list[str]] = {
    NextAction.CONTINUE_SAME_IDEA: ["3.8"],
    NextAction.SWITCH_IDEA: ["3.2", "3.4"],
    NextAction.REPLAN_EXPERIMENT: ["3.5", "3.6_3.7", "3.8"],
    NextAction.STOP: [],
}

class NextRunProposal(BaseModel):
    proposal_id: str
    action: NextAction
    target_stage: Literal["3.2", "3.4", "3.5", "3.6_3.7", "3.8"] | None
    requires_child_run: bool
    reason: str
    prerequisite_actions: list[str]
    evidence_refs: list[ArtifactReferenceV2]

    @model_validator(mode="after")
    def _stage_is_legal_for_action(self):
        legal = _LEGAL_ACTION_STAGES.get(self.action, [])
        if self.action == NextAction.STOP:
            if self.target_stage is not None:
                raise ValueError("STOP requires target_stage=None")
        else:
            if self.target_stage is None:
                raise ValueError(f"action={self.action} requires target_stage")
            if self.target_stage not in legal:
                raise ValueError(
                    f"action={self.action} cannot target stage={self.target_stage}; "
                    f"legal stages: {legal}"
                )
        return self

class ReportFacts(BaseModel):

    resource_comparison: ResourceComparisonReport | None
    artifact_registry: list[ArtifactReferenceV2]
    evidence_sufficiency: dict[str, EvidenceSufficiency]

    @model_validator(mode="after")
    def _baseline_source_consistency(self):
        if self.baseline_source_type == "current_run":
            if self.baseline_source_run_id is not None:
                raise ValueError("current_run must not set baseline_source_run_id")
            if self.baseline_identity_validation_ref is not None:
                raise ValueError("current_run must not set baseline_identity_validation_ref")
        elif self.baseline_source_type == "reused":
            if self.baseline_source_run_id is None:
                raise ValueError("reused requires baseline_source_run_id")
            if self.baseline_identity_validation_ref is None:
                raise ValueError("reused requires baseline_identity_validation_ref")
        return self
```

1. IDEA: idea_label + baseline → variant → idea_support
2. COMPARISON: 每 metric 一张表（PairedMetricObservation → Markdown table）
3. SCIENTIFIC VALIDITY: validity_aggregate
4. REPRODUCIBILITY: reproducibility_report 摘要
5. FAILURE ANALYSIS（如果 failure_analysis 不为 None）
6. RESOURCE: baseline-vs-variant per-variant deltas + budget_status（如果 resource_comparison 不为 None）
7. REFLECTION: key_learnings + next_run_proposals
8. APPENDIX: execution unit ledger
