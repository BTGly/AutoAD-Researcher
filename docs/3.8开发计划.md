## 1.2 WorkspaceExecutionRef（3.8 消费，含条件 Validator）

```python
class WorkspaceExecutionRef(BaseModel):
    workspace_id: str
    subject_type: Literal["baseline", "variant"]
    variant_ids: list[str]

    repository_fingerprint: str
    repository_commit: str
    patch_diff_sha256: str | None
    local_validation_report_sha256: str | None
    patch_runner_handoff_ref: ArtifactRef | None

    @model_validator(mode="after")
    def _validate_subject_consistency(self):
        if self.subject_type == "baseline":
            if self.variant_ids:
                raise ValueError("baseline workspace requires variant_ids=[]")
            if self.patch_diff_sha256 is not None:
                raise ValueError("baseline workspace requires patch_diff_sha256=None")
            if self.local_validation_report_sha256 is not None:
                raise ValueError("baseline workspace requires local_validation_report=None")
            if self.patch_runner_handoff_ref is not None:
                raise ValueError("baseline workspace requires patch_runner_handoff_ref=None")
        else:
            if not self.variant_ids:
                raise ValueError("variant workspace requires non-empty variant_ids")
            if self.patch_diff_sha256 is None:
                raise ValueError("variant workspace requires patch_diff_sha256")
            if self.local_validation_report_sha256 is None:
                raise ValueError("variant workspace requires local_validation_report_sha256")
            if self.patch_runner_handoff_ref is None:
                raise ValueError("variant workspace requires patch_runner_handoff_ref")
        return self
```

## 2.1 ExecutionArtifactBinding + ExecutionArtifactProduction

```python
class ExecutionArtifactBinding(BaseModel):
    """精确输入绑定（来自 3.5 MatrixInputBinding）。"""
    consumer_unit_id: str
    consumer_requirement_id: str
    producer_unit_id: str
    producer_production_id: str
    resolved_artifact_ref: ArtifactRef | None
    resolved_sha256: str | None

    @model_validator(mode="after")
    def _validate_resolved_state(self):
        r = self.resolved_artifact_ref is not None
        s = self.resolved_sha256 is not None
        if r != s:
            raise ValueError("resolved_artifact_ref and resolved_sha256 must be both None or both set")
        return self

class ExecutionArtifactProduction(BaseModel):
    """生产单元的输出产物（保留 3.5 production_id）。"""
    production_id: str
    artifact_type: str
    description: str
    resolved_artifact_ref: ArtifactRef | None  # 执行前=None
    resolved_sha256: str | None               # 执行前=None
```

Validator 检查：
```
每个 consumer.producer_production_id
必须存在于 producer unit.expected_output_artifacts[*].production_id
```

## 2.3 AttemptIdentitySnapshot + AttemptRecord + RetryIdentity

```python
class AttemptIdentitySnapshot(BaseModel):
    execution_unit_plan_sha256: str
    command_sha256: str
    input_refs_sha256: str
    workspace_repository_fingerprint: str

class RetryIdentity:
    """Retry 身份匹配检查。"""
    @staticmethod
    def matches(a: AttemptIdentitySnapshot, b: AttemptIdentitySnapshot) -> bool:
        return a == b

class AttemptRecord(BaseModel):
    attempt_id: str; attempt_index: int
    identity: AttemptIdentitySnapshot
    execution_result_ref: ArtifactRef | None
    metrics_report_ref: ArtifactRef | None
    validity_report_ref: ArtifactRef | None
    resource_usage_ref: ArtifactRef | None
    started_at: datetime; completed_at: datetime | None
```

Retry 决策：
```
RetryIdentity.matches(attempt_01.identity, attempt_02.identity)
→ retry_same_command
不匹配 → do_not_retry
```
